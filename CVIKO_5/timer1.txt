
#include <xc.h>
#include <pic18f46k22.h>
// REV TIMER
#pragma config FOSC = HSMP      // Oscillator Selection bits (HS oscillator (medium power 4-16 MHz))
#pragma config PLLCFG = OFF     // 4X PLL Enable (Oscillator OFF)
#pragma config WDTEN = OFF      // Watchdog Timer Enable bits (Watch dog timer is always disabled. SWDTEN has no effect.)


#define _XTAL_FREQ 8E6              // definice fosc pro knihovnu
#define LED LATDbits.LATD2          // ledka
#define DELAY1    __delay_ms(200);
#define DELAY2    __delay_ms(400);         

//roz?i?te Ukázku 1 tak, aby po stisknutí tla?ítka zm?nila rychlost blikání. Definujte dvojici symbolických konstant.
void init(){

    TRISDbits.RD2 = 0;
    TRISCbits.RC0 = 1;
    
    
    T1CONbits.TMR1CS1 = 0;
    T1CONbits.TMR1CS0 = 0;
    T1CONbits.T1CKPS = 0b11;
    TMR1ON = 1;
}

void main(void) {
    
    init();
    
    while(1){
        
        if(PORTCbits.RC0){
            __delay_ms(50);
            
            LED ^= 1 ;
            DELAY1;
        }
        else{
        
            LED ^= 1 ;
            DELAY2;
        }
        
    }
    
    return;
}
